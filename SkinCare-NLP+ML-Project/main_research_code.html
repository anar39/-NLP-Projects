Introduction¶

Amazon review data provides access to a wealth of textual information related to consumer interests and opinions. In addition to collecting brand liking metrics from customers based on their product ratings, we believe it is possible to build a recommendation system specifically for skincare products (but potentially applicable to other sectors as well) by clustering products based on skin concerns and overall consumer sentiment.

Lemmatization and Punctuation Removal¶

After removing stopwords, the product descriptions and reviews are lemmatized to keep the simplest form of each word. Punctuation is also removed during this process.

In [ ]:
stop_words = set(spacy.lang.en.stop_words.STOP_WORDS)

def preprocess_text(text):
    # Remove punctuations
    text = text.translate(str.maketrans('', '', string.punctuation)) if text else ''
    text = text.replace('br', '') if text else ''
    
    # Remove numbers
    text = ''.join(char for char in text if not char.isdigit()) if text else ''
    
    # Process the text using spaCy
    doc = nlp(text) if text else ''
    
    # Lemmatize tokens and remove stop words
    lemmatized_tokens = [token.lemma_ for token in doc if not token.is_stop] if doc else []
    
    # Join the lemmatized tokens back into text
    lemmatized_text = ' '.join(lemmatized_tokens) if lemmatized_tokens else ''
    
    return lemmatized_text

def display_topics(model, feature_names, no_top_words, topic_names=None):
    for ix, topic in enumerate(model.components_):
        if not topic_names or not topic_names[ix]:
            print("\nTopic ", ix)
        else:
            print("\nTopic: '",topic_names[ix],"'")
        print(", ".join([feature_names[i]
                        for i in topic.argsort()[:-no_top_words - 1:-1]]))

def save_topic_words(model, feature_names, no_top_words, topic_names=None):
    topic_words = {}

        lst = [feature_names[i]

def topic_map_score(x, topic_words):
    review_words = x['lemmatized_review_text'].split()
    topic_count = sum(1 for word in review_words if word in topic_words)
    total_words = len(topic_words)
    
    if topic_count > 0:
        relevance_score = topic_count / total_words
        sentiment_weight = 1 + x['sentiment_score']  
        return relevance_score * sentiment_weight
    else:
       return 0


def topic_map_product(x, topic_words, threshold=0.1):
    review_words = x['lemmatized_review_text'].split()
    topic_count = sum(1 for word in review_words if word in topic_words)
    total_words = len(topic_words)
    
    if total_words > 0:
        relevance_score = topic_count / total_words
        return 1 if relevance_score >= threshold else 0
    else:
        return 0

def recommender(df, concern, n, product_type=None):
    concern_score = concern + '_score'  
    if product_type is None:
        filtered_df = df[(df[concern_score] > 0)]
        filtered_df['combined_score'] = filtered_df[concern_score]
        top_n_products = filtered_df.nlargest(n, 'combined_score')[['product title', concern_score, 'average_rating', 'parent_asin']]
    else:
        filtered_df = df[(df[concern_score] > 0) & (df[product_type] == 1)]
        filtered_df['combined_score'] = filtered_df[concern_score] * filtered_df[product_type]
        top_n_products = filtered_df.nlargest(n, 'combined_score')[['product title', concern_score, product_type, 'average_rating', 'parent_asin']]
    
    if filtered_df.empty:
        filtered_df = df[df[concern_score] > 0]     
    
    
    return top_n_products

def create_topic_graph(topic_dict, top_words=10):
    G = nx.Graph()
    colors = ["#F4744C", "#5275CD", "#FAC958", "#F1A4A3", "#5AA45C", "#FF6347", "#E35982", "#2B9582", "#FFA24B"]

    # adding nodes
    for topic_id, (topic_name, words) in enumerate(topic_dict.items()):
        G.add_node(topic_name, size=len(words), color=colors[topic_id % len(colors)], shape="circle", physics=False,
                   title=topic_name, label=topic_name, font={"size": 20, "color": "white"})

        #  top words as separate nodes connected to the topic node
        for word in words[:top_words]:
            G.add_node(word, size=10, color=colors[topic_id % len(colors)], shape="box", physics=False)
            G.add_edge(topic_name, word, length=200)

    # connecting each topic node to the most similar other topic node
    for topic1_id, (topic1_name, words1) in enumerate(topic_dict.items()):
        max_common_words = 0
        most_similar_topic = None
        for topic2_id, (topic2_name, words2) in enumerate(topic_dict.items()):
            if topic1_name != topic2_name:
                common_words = len(set(words1[:100]) & set(words2[:100]))
                if common_words > max_common_words:
                    max_common_words = common_words
                    most_similar_topic = topic2_name
        if most_similar_topic is not None:
            G.add_edge(topic1_name, most_similar_topic, length=500, value=2, color = 'lightgrey')

    return G

def preprocess_text_w_qst(text):
    # Tokenize the text
    tokens = nltk.word_tokenize(text)
    
    # Remove non-alphabetic characters except '?'
    words = [word for word in tokens if word.isalpha() or '?' in word]
    
    # Lemmatize the words
    lemmatizer = WordNetLemmatizer()
    lemmatized_words = [lemmatizer.lemmatize(word) for word in words]
    
    # Join the lemmatized words back into a full text
    processed_text = ' '.join(lemmatized_words)
    
    return processed_text

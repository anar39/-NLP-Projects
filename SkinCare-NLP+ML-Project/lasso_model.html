Lasso model using Amazon reviews after lemmatization:

# Load Amazon reviews file
data_amazon<-read_csv("reviews_df.csv")
## Rows: 86614 Columns: 14
## ── Column specification ────────────────────────────────────────────────────────
## Delimiter: ","
## chr  (8): review title, review_text, review images, asin, user_id, tokenized...
## dbl  (3): rating, helpful_vote, word_count
## lgl  (2): author, verified_purchase
## dttm (1): timestamp
## 
## ℹ Use `spec()` to retrieve the full column specification for this data.
## ℹ Specify the column types or set `show_col_types = FALSE` to quiet this message.
# Clean up lemmatized_review and create a new column lemmatized_review_text
data_amazon$lemmatized_review_text <- str_replace_all(data_amazon$lemmatized_review, "['\\[\\]]", "") %>%
                                      str_replace_all("'", "") %>%
                                      str_replace_all(", ", " ") %>%
                                      str_replace_all("\"", "")

The first step before building the Lasso model was to split our Amazon reviews’ data into training (80%) and test sets (20%).

# Set seed for reproducibility
set.seed(123)

# Split data into training (80%) and test (20%) sets
train_index <- createDataPartition(data_amazon$rating, p = 0.8, list = FALSE)
training_data_amazon <- data_amazon[train_index, ]
test_data_amazon <- data_amazon[-train_index, ]

Then, we ran the TAB_dfm function on the lemmatized_review_text, which tokenised the review text, removed stop words, and returned a document feature matrix. Then, we constructed the Lasso model using only unigrams as this gave the highest accuracy compared to bigrams or trigrams.

# LASSO model using unigrams
data_dfm_train_amazon <- TAB_dfm(training_data_amazon$lemmatized_review_text, ngrams = 1)
data_dfm_test_amazon <- TAB_dfm(test_data_amazon$lemmatized_review_text, ngrams = 1, min.prop=0) %>%
  dfm_match(colnames(data_dfm_train_amazon))

data_model_amazon<-glmnet::cv.glmnet(x=data_dfm_train_amazon %>%
                               as.matrix(),y=training_data_amazon$rating)

plot_data_model <- plot(data_model_amazon)

saveRDS(data_dfm_train_amazon, "function_data_dfm_train_amazon.rds")
saveRDS(data_dfm_test_amazon, "function_data_dfm_test_amazon.rds")
saveRDS(data_model_amazon, "function_data_model_amazon.rds")

Finally, we built a coefficients plot to visually interpret which words in the review text are the primary indicators of the rating (1 to 5 stars).

# Coefficients plot
plotDat_amazon <- data_model_amazon %>%
  coef() %>%
  drop() %>%
  as.data.frame() %>%
  rownames_to_column(var = "ngram") %>%
  rename(score = ".") %>%
  filter(score != 0 & ngram != "(Intercept)" & !is.na(score))  %>%
  left_join(data.frame(ngram = colnames(data_dfm_train_amazon),
                       freq = colMeans(data_dfm_train_amazon)))

plotDat_amazon %>%
  mutate_at(vars(score, freq), ~round(., 3)) %>%
  ggplot(aes(x = score, y = freq, label = ngram, color = score)) +
  scale_color_gradient(low = "#F4744C", high = "#305DD0") +
  geom_vline(xintercept = 0) +
  geom_point() +
  geom_label_repel(max.overlaps = 45, force = 6) +  
  scale_y_continuous(trans = "log2", breaks = c(-5, 0.01, 0.05, 0.1, 0.2, 0.5, 1, 2, 5)) +
  scale_x_continuous(limits = c(-1, 1)) +
  theme_bw() +
  labs(x = "Coefficient in Model", y = "Uses per Amazon Review") +
  theme(legend.position = "none",
        axis.title = element_text(size = 15),
        axis.text = element_text(size = 15))

Transfer learning (use Amazon model to predict Sephora rating):

In order to understand how well our model can be generalized to other skin care datasets, we applied some transfer learning and used our Lasso model to predict Sephora ratings.
